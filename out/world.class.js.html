class World {
    ctx;
    level1 = level1;
    backgroundObjectsHell = level1.backgroundObjectsHell;
    backgroundObjects = level1.backgroundObjects;
    tileObjectsHell = level1.tileObjectsHell;
    tileObjects = level1.tileObjects;
    birds = level1.birds;
    character = new Character();
    enemies = level1.enemies;
    projectiles = level1.projectiles;
    hearts = level1.hearts;
    screens = level1.screens;

    hpBarBackground = healthBar.barBackground;
    hpPoints = healthBar.point;
    hpBarFront = healthBar.barFront;

    manaBarBackground = manaBar.barBackground;
    manaPoints = manaBar.point;
    manaBarFront = manaBar.barFront;

    canvas;
    keyboard;
    camera_x = 0;
    amountOfDeadEnemys = 0;
    bossSpawned = false;
    characterIsFalling = false;

    collect_heart_sound = new Audio('./audio/collect-heart.mp3');

    constructor(canvas, keyboard) {
        this.ctx = canvas.getContext('2d');
        this.canvas = canvas;
        this.keyboard = keyboard;
        this.draw();
        this.setWorld();
        this.checkCollisions();
        this.createEnemy();
    }

    setWorld() {
        this.character.world = this;
    }

    /**
     * starts intervals that check for character colliding with an enemy or heart 
     */
    checkCollisions() {
        this.checkEnemyCollision();
        this.checkHeartCollision();
    }

    /**
     * interval that check character if character is colliding with an enemy
     */
    checkEnemyCollision() {
        setInterval(() => {
            this.enemies.forEach((enemy) => {
                if (this.character.isColliding(enemy)) {
                    enemy.isColliding = true;
                    let jumpOnEnemy = this.checkHeightOnCollision();
                    this.checkForEnemyAttack(enemy, jumpOnEnemy);
                    this.checkForCharacterJumpOnEnemy(enemy, jumpOnEnemy);
                } else {
                    enemy.isColliding = false;
                }
            });
        }, 5);
    }

    /**
     * check if colliding enemy is able to attack character
     * @param {object} enemy 
     * @param {boolean} jumpOnEnemy 
     */
    checkForEnemyAttack(enemy, jumpOnEnemy) {
        if (enemy.attackAnimationStarted == false && jumpOnEnemy == false && enemy.dyingAnimationStarted == false && this.character.dyingAnimationStarted == false && enemy.hurtAnimationStarted == false && enemy.fallingDown == false && world.characterIsFalling == false) {
            enemy.attackAnimationStarted = true;
            enemy.attack(enemy.id);
            playSound(this.character.hurt_sound, false);
            playSound(enemy.chop_sound, false)
            this.character.hurt(10);
        }
    }

    /**
     * check if colliding enemy is hit by character
     * @param {object} enemy 
     * @param {boolean} jumpOnEnemy 
     */
    checkForCharacterJumpOnEnemy(enemy, jumpOnEnemy) {
        if (enemy.dyingAnimationStarted == false && jumpOnEnemy == true && enemy.hurtAnimationStarted == false && world.characterIsFalling == false) {
            enemy.hurt(enemy.id, 50);
            playSound(this.character.jump_on_enemy_sound, false)
            this.character.jumpAnimationCounter = 0;
            this.character.highJumpAnimationCounter = 0;
            this.character.highJumpAnimationSet = true;
            this.character.jump();
        }
    }

    /**
     * check each heart for colliding with character
     */
    checkHeartCollision() {
        setInterval(() => {
            this.hearts.forEach(heart => {
                if (this.character.isColliding(heart)) {
                    playSound(this.collect_heart_sound, false);
                    this.addHealthPoints();
                    this.removeHeart(heart);
                }
            });
        }, 5);
    }

    /**
     * add healthpoints to character and fill health to maximum if lost health of character is smaller then the default amount of health regenerated by a heart
     */
    addHealthPoints() {
        if (world.character.health >= 57) {
            world.character.health = 87;
        } else {
            world.character.health = world.character.health + 30;
        }
    }

    /**
     * remove heart object
     */
    removeHeart(heart) {
        for (let i = 0; i < world.hearts.length; i++) {
            if (heart === world.hearts[i]) {
                world.hearts.splice(i, 1);
            }
        }
    }

    /**
     * check if character is in the air when colliding with an enemy
     * @returns boolean
     */
    checkHeightOnCollision() {
        if (this.character.y <= 250 && this.character.speedY < 0) {
            return true;
        } else {
            return false;
        }
    }

    /**
     * draw all objects on canvas
     */
    draw() {
        this.ctx.clearRect(0, 0, canvas.width, canvas.height)

        this.ctx.translate(this.camera_x, 0);

        this.addObjectToMap(this.backgroundObjectsHell);
        this.addObjectToMap(this.backgroundObjects);
        this.addObjectToMap(this.tileObjectsHell);
        this.addObjectToMap(this.tileObjects);
        this.addObjectToMap(this.birds);
        this.addObjectToMap(this.hearts);
        this.addToMap(this.character);
        this.addObjectToMap(this.enemies);
        this.addObjectToMap(this.projectiles);

        this.ctx.translate(-this.camera_x, 0);
        this.addObjectToMap(this.hpBarBackground);
        this.addObjectToMap(this.hpPoints);
        this.addObjectToMap(this.hpBarFront);
        this.addObjectToMap(this.manaBarBackground);
        this.addObjectToMap(this.manaPoints);
        this.addObjectToMap(this.manaBarFront);
        this.addObjectToMap(this.screens);
        this.ctx.translate(this.camera_x, 0);



        this.ctx.translate(-this.camera_x, 0);

        let self = this;
        requestAnimationFrame(function () {
            self.draw();
        });
    }

    /**
     * add object to map
     * @param {object} objects 
     */
    addObjectToMap(objects) {
        objects.forEach(o => {
            this.addToMap(o);
        });
    }

    /**
     * check if image has to be flipped
     * @param {object} mo 
     */
    addToMap(mo) {
        if (mo.otherDirection) {
            this.flipImage(mo);
        }

        mo.draw(this.ctx);

        if (mo.otherDirection) {
            this.flipImageBack(mo);
        }
    }

    /**
     * sets interval that create a new Object of Enemy 
     */
    createEnemy() {
        setInterval(() => {
            if (this.bossSpawned == false && this.amountOfDeadEnemys >= 3) {
                this.bossSpawned = true;
                this.createBoss();
                this.changeWorld();
            } else if ((world.enemies.length <= 4) && this.bossSpawned == false) {
                this.createOrc();
            }
        }, 5000);
    }

    /**
     * timeout that creates a new Object of EnemyBoss
     */
    createBoss() {
        setTimeout(() => {
            let newOrc = new EnemyBoss;
            this.enemies.push(newOrc);
        }, 3000);
    }

    /**
     * create a new Object of EnemyOne() or EnemyTwo() depending on a random mathematical value 
     */
    createOrc() {
        let rng = Math.random();
        let newOrc;

        if (rng < 0.5) {
            newOrc = new EnemyOne();
        } else {
            newOrc = new EnemyTwo();
        }
        this.enemies.push(newOrc);
    }

    /**
     * starts all necessary functions that are relevant for changing terrain
     */
    changeWorld() {
        this.makeCharacterFalling();
        this.makeEnemysFalling();
        this.moveBackgroundObjects();
        this.moveHellBackgroundObjects();
        this.moveHellTileObjects();
        this.moveTileObjects();
    }

    /**
     * load falling image
     */
    makeCharacterFalling() {
        this.characterIsFalling = true;
        this.character.loadImage('./img/Mage/Jump/jump6.png');
    }

    /**
     * load falling image for all enemys
     */
    makeEnemysFalling() {
        this.enemies.forEach(enemy => {
            if (enemy.enemyIsDead == false) {
                enemy.fallingDown = true;
                enemy.loadImage(enemy.fallingDownImage);
            }
        });
    }

    /**
     * move all background Objects
     */
    moveBackgroundObjects() {
        this.backgroundObjects.forEach((backgroundObject) => {
            setInterval(() => {
                backgroundObject.y -= 6;
            }, 20);
        });
    }

    /**
     * move all hell background objects
     */
    moveHellBackgroundObjects() {
        this.backgroundObjectsHell.forEach((backgroundObject) => {
            let moveNewBackground = setInterval(() => {
                backgroundObject.y -= 6;
                this.checkHellBackgroundForFinalPosition(moveNewBackground, backgroundObject);
            }, 20);
        });
    }

    /**
     * check hell background for final position and adjust deviations 
     * @param {object} moveNewBackground 
     */
    checkHellBackgroundForFinalPosition(moveNewBackground, backgroundObject) {
        if (backgroundObject.y < 0) {
            backgroundObject.y = 0;
            clearInterval(moveNewBackground);
        }
    }

    /**
     * check hell background tile objects for final position and adjust deviations
     */
    moveHellTileObjects() {
        this.tileObjectsHell.forEach((backgroundObject) => {
            let moveNewBackground = setInterval(() => {
                backgroundObject.y -= 6;
                this.checkBackgroundFinalPosition(moveNewBackground, backgroundObject);
            }, 20);
        });
    }

    /**
     * check background for final position and adjust deviations
     * load character standing image
     * @param {setInterval} moveNewBackground 
     */
    checkBackgroundFinalPosition(moveNewBackground, backgroundObject) {
        if (backgroundObject.y < 420) {
            backgroundObject.y = 420;
            this.character.loadImage('./img/Mage/Idle/idle1.png');
            this.characterIsFalling = false;
            this.resetEnemyFallingDown();
            clearInterval(moveNewBackground);
        }
    }

    /**
     * reset falling down animation for each enemy
     */
    resetEnemyFallingDown() {
        this.enemies.forEach(enemy => {
            enemy.fallingDown = false;
            if (enemy.enemyIsDead == false) {    
            enemy.loadImage(enemy.fallingDownImage);
            }
        });
    }

    /**
     * move all tile Objects after a random time to maximum of 500ms
     */
    moveTileObjects() {
        this.tileObjects.forEach((TileObject) => {
            let timeout = Math.random() * 500;
            setTimeout(() => {
                setInterval(() => {
                    TileObject.y += 2;
                }, 20);
            }, timeout);
        });
    }

    /**
     * flip image
     * @param {object} mo 
     */
    flipImage(mo) {
        if (mo instanceof Character) {
            this.ctx.save();
            this.ctx.translate(mo.width - 50, 0);
            this.ctx.scale(-1, 1);
            mo.x = mo.x * -1;
        } else {
            this.ctx.save();
            this.ctx.translate(mo.width, 0);
            this.ctx.scale(-1, 1);
            mo.x = mo.x * -1;
        }
    }

    /**
     * flip image back
     * @param {object} mo 
     */
    flipImageBack(mo) {
        mo.x = mo.x * -1;
        this.ctx.restore();
    }
}
